# 请你判断一个 9 x 9的数独是否有效

```js
/*
 请你判断一个 9 x 9的数独是否有效
 1. 数字1-9 在每一行只能出现一次
 2. 数字1-9 在每一列只能出现一次
 3. 数字 1-9 在每一个以粗实线分隔的3x3 宫内只能出现一次。
 */

const board1 = [
    ["5", "3", ".", ".", "7", ".", ".", ".", "."],
    ["6", ".", ".", "1", "9", "5", ".", ".", "."],
    [".", "9", "8", ".", ".", ".", ".", "6", "."],
    ["8", ".", ".", ".", "6", ".", ".", ".", "3"],
    ["4", ".", ".", "8", ".", "3", ".", ".", "1"],
    ["7", ".", ".", ".", "2", ".", ".", ".", "6"],
    [".", "6", ".", ".", ".", ".", "2", "8", "."],
    [".", ".", ".", "4", "1", "9", ".", ".", "5"],
    [".", ".", ".", ".", "8", ".", ".", "7", "9"]
]

const board2 = [
    ["8", ".", "7", ".", ".", ".", ".", ".", "."],
    [".", "1", ".", ".", ".", ".", ".", ".", "."],
    ["8", ".", ".", ".", ".", ".", ".", ".", "."],
    [".", ".", ".", ".", ".", ".", ".", ".", "."],
    [".", ".", ".", ".", ".", ".", ".", ".", "."],
    [".", ".", ".", ".", ".", ".", ".", ".", "."],
    [".", ".", ".", ".", ".", ".", ".", ".", "."],
    [".", ".", ".", ".", ".", ".", ".", ".", "."],
    [".", ".", ".", ".", ".", ".", ".", ".", "."]
]

const isValidSudoku = function (board) {
    // 创建三个哈希表，分别用于记录每行、每列和每个3x3宫内数字的出现情况
    const rows = new Array(9).fill(0).map(() => new Map());
    const columns = new Array(9).fill(0).map(() => new Map());
    const boxes = new Array(9).fill(0).map(() => new Map());

    // 遍历数独的每个格子
    for (let i = 0; i < 9; i++) {
        for (let j = 0; j < 9; j++) {
            const num = board[i][j];

            if (num !== '.') {
                // 计算当前数字在哪个3x3宫内
                const boxIndex = Math.floor(i / 3) * 3 + Math.floor(j / 3);

                // 检查当前数字在行、列和3x3宫内是否已经出现过
                if (rows[i].has(num) || columns[j].has(num) || boxes[boxIndex].has(num)) {
                    return false; // 如果已经出现过，则数独无效
                }

                // 将当前数字添加到对应的哈希表中
                rows[i].set(num, 1);
                columns[j].set(num, 1);
                boxes[boxIndex].set(num, 1);
            }
        }
    }

    return true; // 如果所有数字都符合规则，则数独有效
};

console.log(isValidSudoku(board1));
console.log(isValidSudoku(board2));
```
